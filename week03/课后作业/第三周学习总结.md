```python
class Recursive():
  def recursive(self, level, param1, param2...):
    # 递归终止条件
    if level > max_level:
      process_result()
      return
    # 当层处理
    process(level, data...)
    #向下递归
    self.recursive(level+1, p1, p2...)
    #递归后处理，还原
```
